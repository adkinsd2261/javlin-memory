function memoryOS() { return { // Core state totalMemories: 0, efficiencyScore: 0, systemStatus: 'initializing', sessionId: '', currentFocus: 'System starting...', successRate: '0%', categories: {}, nextSteps: 'Loading system data...', // Chat functionality chatHistory: [], currentMessage: '', isTyping: false, async init() { console.log('ðŸ§  MemoryOS initializing...'); await this.loadSystemData(); this.setupPeriodicUpdates(); }, async loadSystemData() { try { // Load stats const statsResponse = await fetch('/stats'); const stats = await statsResponse.json(); this.totalMemories = stats.total_memories || 0; this.successRate = stats.success_rate || '0%'; this.categories = stats.categories || {}; // Calculate efficiency score const successPercent = parseFloat(this.successRate.replace('%', '')); this.efficiencyScore = Math.min(Math.max(successPercent, 0), 100); // Load GPT status const gptResponse = await fetch('/gpt-status'); const gptData = await gptResponse.json(); this.systemStatus = gptData.system_status === 'online' ? 'operational' : 'degraded'; this.sessionId = Date.now().toString(); this.currentFocus = gptData.message || 'Ready'; this.nextSteps = 'System operational'; } catch (error) { console.error('Failed to load system data:', error); this.systemStatus = 'error'; this.currentFocus = 'System error'; this.nextSteps = 'Check system health'; } }, setupPeriodicUpdates() { // Update system data every 30 seconds setInterval(() => { this.loadSystemData(); }, 30000); }, async sendMessage() { if (!this.currentMessage.trim() || this.isTyping) return; const userMessage = this.currentMessage.trim(); this.chatHistory.push({ id: Date.now(), type: 'user', content: userMessage, timestamp: new Date().toLocaleTimeString() }); this.currentMessage = ''; this.isTyping = true; try { // Send to AI endpoint const response = await fetch('/ai/query', { method: 'POST', headers: {'Content-Type': 'application/json'}, body: JSON.stringify({input: userMessage}) }); const result = await response.json(); this.chatHistory.push({ id: Date.now() + 1, type: 'assistant', content: result.ai_response?.response || result.error || 'No response', timestamp: new Date().toLocaleTimeString() }); } catch (error) { this.chatHistory.push({ id: Date.now() + 2, type: 'error', content: 'Failed to get AI response', timestamp: new Date().toLocaleTimeString() }); } finally { this.isTyping = false; this.$nextTick(() => { const container = this.$refs.chatContainer; if (container) { container.scrollTop = container.scrollHeight; } }); } } }; }
